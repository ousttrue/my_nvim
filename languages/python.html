DebugAdapter¶

$ pip install debugpy

local dap = require("dap")
dap.adapters.python = {
    type = "executable",
    command = "C:/Python38/python.exe",
    args = { "-m", "debugpy.adapter" },
}
dap.configurations.python = {
    {
        type = "python",
        request = "launch",
        name = "Launch file",
        program = "${file}",
        pythonPath = function()
            return "C:/Python38/python.exe"
        end,
    },
}
